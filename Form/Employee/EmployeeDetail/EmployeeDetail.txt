ID ["+vT_employeeUUID+"]")		Else 			DBG_Log ("Existing employee. ID ["+vT_employeeUUID+"]")		End if 				If (False)  // Not behaving :(			C_POINTER($vP_CancelBtn;$vP_OKBtn)			C_LONGINT($vL_tempForeColour;$vL_tempBackColour)						$vP_CancelBtn:=OBJECT Get pointer(Object named;"empDetailCancel_Btn")			If (Nil($vP_CancelBtn))				DBG_Log ("Could not get pointer for cancel button")			Else 				OBJECT SET COLOR($vP_CancelBtn->;(White+(256*Orange)))			End if 						$vP_OKBtn:=OBJECT Get pointer(Object named;"empDetailOK_Btn")			If (Nil($vP_OKBtn))				DBG_Log ("Could not get pointer for ok button")			Else 				  // OBJECT SET COLOR($vP_OKBtn->;Orange)				OBJECT SET COLOR($vP_CancelBtn->;(White+(256*Orange)))			End if 		End if 				C_POINTER($vP_recNum)		$vP_recNum:=OBJECT Get pointer(Object named;"recNum_Lbl")		If (Nil($vP_recNum))			DBG_Log ("Could not get pointer for record number label")		Else 			OBJECT SET VERTICAL ALIGNMENT($vP_recNum;Align center)		End if 				  //-----------------------------------------------------------------------------	: ($vL_formEvent=On Clicked)		Case of 			: (vB_empDetailOKBtn=1)				DBG_Log ("OK clicked, check before and after entries")				SAVE RECORD([Employee])				ACCEPT							: (vB_empDetailCancelBtn=1)				DBG_Log ("Employee detail cancelled")				vT_employeeUUID:=""				CANCEL						End case 				  //-----------------------------------------------------------------------------	: ($vL_formEvent=On Data Change)		DBG_Log ("Data change event")		DBG_Log ("Records in selection([Employee]) = ["+\			String(Records in selection([Employee]))+"] == 1 -> ["+\			String(Records in selection([Employee])=1)+"]")		DBG_Log ("         [Employee]ID = ["+[Employee]ID+"]")		DBG_Log ("    [Employee]Surname = ["+[Employee]Surname+"]")		DBG_Log ("  [Employee]FirstName = ["+[Employee]FirstName+"]")		DBG_Log ("[Employee]IsManagerFg = ["+String([Employee]IsManagerFg)+"]")		DBG_Log ("  [Employee]StartDate = ["+String([Employee]StartDate)+"]")				  //-----------------------------------------------------------------------------		  // Moar to come		End case DBG_EndMethod
