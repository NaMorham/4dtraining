  //%attributes = {"lang":"en"} comment added and reserved by 4D.  //Project Method: LBL_AlignAndPopulateObject($vT_objName;$vT_newText) -> String  //  //   Parameter        Type             Description  //    $vT_objName     Text     <-      The name of the object to align and update  //    $vT_newText     Text     <-      The new text to display in the object  //    $vT_oldText     Text     ->      The text from the object before update  //  //Description:  //   Vertically center and update text of a label  //  //Date        Init  Description  //===============================================================================  //17/05/2017   AH   Initial version.DBG_BeginMethod (Current method name)C_TEXT($1;$vT_objName)C_TEXT($2;$vT_newText)C_TEXT($0;$vT_oldText)C_POINTER($vP_labelObj)$vT_objName:=$1$vT_newText:=$2$vP_labelObj:=OBJECT Get pointer(Object named;$vT_objName)If (Nil($vP_labelObj))	$vT_oldText:=""	DBG_Log ("Could not get object pointer for label \""+$vT_objName+"\"")Else 	OBJECT SET VERTICAL ALIGNMENT($vP_labelObj;Align center)	$vT_oldText:=$vP_labelObj->	$vP_labelObj->:=$vT_newTextEnd if $0:=$vT_oldTextDBG_EndMethod
