  //Project Method: HOST_MCSU_SYS_GetDBParam_S(Selector{;StrValuePtr}) -> Value
  //
  //   Parameter        Type             Description
  //   Selector         Longint     ->   The 4D database parameter for which to
  //                                     retrieve the value.
  //   StrValuePtr      Pointer     <-   Pointer to a string variable in which to
  //                                     return selectors with string values.
  //   Value            Longint     <-   Current value of the selected database
  //                                     parameter.
  //
  //Description
  //   Returns the value of the database parameter for the supplied selector.
  //   Shared with Host & Component so that components can retrieve database
  //   parameter values from their host databases, e.g. Client Server Port ID, etc.
  //   Has "Execute on server" attribute set to enable retrieval directly from
  //   the server.
  //
  //Date        Init  Description
  //===============================================================================
  //09/05/2017   4D   Automatically generated by the MCSU component.

C_LONGINT($0;$vL_Value)
C_LONGINT($1;$vL_Selector)
C_POINTER($2;$vP_StrValue)

C_TEXT($vT_Value)

$vL_Value:=0

$vL_Selector:=$1
If (Count parameters>=2)
	$vP_StrValue:=$2
	$vL_Value:=Get database parameter($vL_Selector;$vT_Value)
	$vP_StrValue->:=$vT_Value
Else
	$vL_Value:=Get database parameter($vL_Selector)
End if

$0:=$vL_Value
