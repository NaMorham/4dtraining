//%attributes = {"lang":"en"} comment added and reserved by 4D.//Project Method: Bool_isManager(vT_employeeId////   Parameter        Type             Description//    vT_employeeId   Text     <-      The employee ID//    vB_result       Bool     ->      True if the employee is a manager////Description://  An employee is considered a manger if they are currently employed and flagged as a manager//   ////Date        Init  Description//===============================================================================//04/05/2017   AH   Initial version.C_TEXT($1;$vT_employeeId)C_BOOLEAN($0;$vB_result)$vT_employeeId:=$1DebugLog ("a) Source selection size = ["+String(Records in selection([Employee]))+"]")// TODO: isUUID method$vB_result:=Bool_isEmployed ($vT_employeeId)If (Not($vB_result))DebugLog ("Employee with ID ["+$vT_employeeId+"] is not an employee")$vB_result:=FalseElse DebugLog ("b) Source selection size = ["+String(Records in selection([Employee]))+"]")// Is this needed? or is the selection kept from the call to isEmployedQUERY([Employee];[Employee]ID=$vT_employeeId)$vB_result:=([Employee]IsManagerFg & ([Employee]StartDate<=Current date))$vB_result:=$vB_result & (([Employee]EndDate>Current date) | ([Employee]EndDate=!00/00/0000!))End if $0:=$vB_result
