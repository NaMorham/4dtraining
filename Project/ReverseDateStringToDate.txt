  //%attributes = {"lang":"en"} comment added and reserved by 4D.  //Project Method: ReverseDateStringToDate(vT_revDateString) -> Date  //  //   Parameter        Type             Description  //   vT_revDateString Text    <-       Date sting in format (YEAR-MONTH-DAY)  //   vD_result        Date    ->       Converted date or 00/00/0000  //  //Description:  //     //  //Date        Init  Description  //===============================================================================  //02/05/2017   AH   Initial version.C_TEXT($1;$vT_revDateString;$vT_forwardStr)C_DATE($0;$vD_result)ARRAY TEXT($aT_tokens;0)C_LONGINT($vL_numTokens)$vT_revDateString:=$1$vL_numTokens:=GK_TextToArray ($vT_revDateString;$aT_tokens;"/-\\ ";kGK_TTA_REMOVE_EMPTY)If ($vL_numTokens#3)  // Must have year month and day	$vD_result:=!00-00-00!Else 	$vT_forwardStr:=$aT_tokens{3}+"-"+$aT_tokens{2}+"-"+$aT_tokens{1}	$vD_result:=Date($vT_forwardStr)End if $0:=$vD_result
